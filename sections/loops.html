5.1 For Loop

For loop digunakan untuk iterasi dengan jumlah perulangan yang sudah diketahui.

Struktur for loop:
for (inisialisasi; kondisi; increment/decrement) {
    // kode yang akan diulang
}
Contoh for loop ascending (1-5):
#include <stdio.h>

int main() {
    for (int i = 1; i <= 5; i++) {
        printf("%d ", i);
    }
    return 0;
}
Contoh for loop descending (5-1):
#include <stdio.h>

int main() {
    for (int i = 5; i >= 1; i--) {
        printf("%d ", i);
    }
    return 0;
}
Latihan for loop ascending 1-100:
#include <stdio.h>

int main() {
    for (int i = 1; i <= 100; i++) {
        printf("%d ", i);
    }
    return 0;
}

5.2 While Loop

While loop digunakan ketika jumlah perulangan belum diketahui, tapi tergantung kondisi.

Contoh while loop:
#include <stdio.h>

int main() {
    int j = 1;
    while (j <= 5) {
        printf("%d ", j);
        j++;
    }
    return 0;
}

5.3 Do-While Loop

Do-while loop mirip dengan while, tapi perulangan dijalankan minimal sekali.

Contoh do-while loop:
#include <stdio.h>

int main() {
    int k = 1;
    do {
        printf("%d ", k);
        k++;
    } while (k <= 5);
    return 0;
}

5.4 Contoh Do-While dengan Switch-Case

Contoh penggunaan do-while loop dengan menu sederhana menggunakan switch-case.

Contoh do-while + switch-case:
#include <stdio.h>

int main() {
    int pilihan;

    do {
        printf("=== Menu Sederhana ===\n");
        printf("1. Cetak Hello\n");
        printf("2. Cetak Dunia\n");
        printf("3. Keluar\n");
        printf("Masukkan pilihan: ");
        scanf("%d", &pilihan);

        switch (pilihan) {
            case 1:
                printf("Hello!\n");
                break;
            case 2:
                printf("Dunia!\n");
                break;
            case 3:
                printf("Keluar dari program.\n");
                break;
            default:
                printf("Pilihan tidak valid!\n");
        }

        printf("\n"); // biar lebih rapi

    } while (pilihan != 3);

    return 0;
}

5.5 Nested Loop

Nested loop adalah perulangan bersarang, yaitu for/while di dalam for/while lainnya.

Contoh pola bintang segitiga:
#include <stdio.h>

int main() {
    for (int i = 1; i <= 5; i++) {
        for (int j = 1; j <= i; j++) {
            printf("* ");
        }
        printf("\n");
    }
    return 0;
}

Latihan Praktek

  • Buat program yang menampilkan bilangan ganjil dari 1 sampai N.
    • Jawaban:
    #include <stdio.h>

int main() {
    int N;
    printf("Masukkan bilangan N: ");
    scanf("%d", &N);

    printf("Bilangan ganjil dari 1 sampai %d:\n", N);
    for (int i = 1; i <= N; i++) {
        if (i % 2 != 0) {
            printf("%d ", i);
        }
    }
    return 0;
}
  • Seorang anak menabung Rp1000 di hari pertama, dan bertambah Rp500 setiap hari berikutnya. Buat program untuk menghitung total tabungan setelah N hari.
    • Jawaban:
    #include <stdio.h>

int main() {
    int N;
    printf("Masukkan jumlah hari: ");
    scanf("%d", &N);

    int total = 0;
    int tabungan = 1000;

    for (int i = 1; i <= N; i++) {
        total += tabungan;
        tabungan += 500;
    }

    printf("Total tabungan setelah %d hari = Rp%d\n", N, total);
    return 0;
}
  • Seorang pelanggan berbelanja N kali. Setiap kali belanja Rp20.000. Jika total belanja mencapai Rp100.000 atau lebih, ia mendapat diskon Rp10.000. Hitung total pembayaran setelah N kali belanja.
    • Jawaban:
    #include <stdio.h>

int main() {
    int N;
    printf("Masukkan jumlah belanja (N kali): ");
    scanf("%d", &N);

    int total = N * 20000;

    if (total >= 100000) {
        total -= 10000;
    }

    printf("Total pembayaran = Rp%d\n", total);
    return 0;
}
