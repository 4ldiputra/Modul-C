Fungsi printf() - Output

Fungsi printf() digunakan untuk menampilkan output ke layar dengan berbagai format.

Format Specifier Tipe Data Contoh
%d atau %i Integer printf("%d", 42);
%f Float printf("%.2f", 3.14);
%c Character printf("%c", 'A');
%s String printf("%s", "Hello");
%x Hexadecimal printf("%x", 255);
Contoh Penggunaan printf():

#include 

int main() {
    int umur = 20;
    float tinggi = 175.5;
    char grade = 'A';
    char nama[] = "Budi";
    
    printf("Nama: %s\n", nama);
    printf("Umur: %d tahun\n", umur);
    printf("Tinggi: %.1f cm\n", tinggi);
    printf("Grade: %c\n", grade);
    printf("Nilai dalam hex: %x\n", umur);
    
    return 0;
}
Contoh Lengkap Input dan Output:

#include 

int main() {
    char nama[20];
    int umur;
    float tinggi;

    // Input
    printf("Masukkan nama: ");
    scanf("%s", nama);
    
    printf("Masukkan umur: ");
    scanf("%d", &umur);

    printf("Masukkan tinggi badan: ");
    scanf("%f", &tinggi);

    // Output
    printf("\n--- Data yang Dimasukkan ---\n");
    printf("Nama   : %s\n", nama);
    printf("Umur   : %d tahun\n", umur);
    printf("Tinggi : %.2f cm\n", tinggi);

    return 0;
}
