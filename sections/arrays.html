6.1 Array

Array adalah struktur data yang digunakan untuk menyimpan kumpulan elemen dengan tipe data yang sama dalam lokasi memori yang berdekatan.

Deklarasi dan Inisialisasi Array

// Deklarasi array
int numbers[5];

// Inisialisasi saat deklarasi
int numbers[5] = {1, 2, 3, 4, 5};

// Inisialisasi sebagian
int numbers[5] = {1, 2}; // sisanya akan diisi 0

Mengakses Elemen Array

int numbers[5] = {10, 20, 30, 40, 50};

// Mengakses elemen (indeks dimulai dari 0)
printf("%d\n", numbers[0]); // Output: 10
printf("%d\n", numbers[4]); // Output: 50

// Mengubah nilai elemen
numbers[2] = 35;

Pemanggilan Array dengan Perulangan

int numbers[5] = {10, 20, 30, 40, 50};

for (int i = 0; i < 5; i++) {
    printf("%d ", numbers[i]);
}
// Output: 10 20 30 40 50

Input dari User ke Array

#include <stdio.h>

int main() {
    int data[5];
    for (int i = 0; i < 5; i++) {
        printf("Masukkan elemen ke-%d: ", i);
        scanf("%d", &data[i]);
    }

    printf("Isi array: ");
    for (int i = 0; i < 5; i++) {
        printf("%d ", data[i]);
    }

    return 0;
}

Menghitung Jumlah Elemen dengan sizeof

#include <stdio.h>

int main() {
    int numbers[5] = {10, 20, 30, 40, 50};
    int panjang = sizeof(numbers) / sizeof(numbers[0]);

    printf("Jumlah elemen array = %d\n", panjang);

    return 0;
}

Mencari Nilai Terbesar dalam Array

#include <stdio.h>

int main() {

        }

    return 0;

Latihan Praktek

  • SOAL 1: Seorang guru ingin menyimpan nilai ujian 5 siswa di dalam array, lalu mencari:
    • Nilai tertinggi.
    • Nilai terendah.
    • Rata-rata nilai siswa.
    Buat program dengan input nilai dari user dan tampilkan hasilnya.

Contoh Program Nilai Siswa

#include <stdio.h>

int main() {
    int nilai[5];
    int maks, min, total = 0;
    float rata;

    // Input nilai siswa
    for (int i = 0; i < 5; i++) {
        printf("Masukkan nilai siswa ke-%d: ", (i + 1));
        scanf("%d", &nilai[i]);
    }

    // Asumsikan nilai awal sebagai acuan
    maks = nilai[0];
    min = nilai[0];

    // Proses mencari maks, min, dan total
    for (int i = 0; i < 5; i++) {
        if (nilai[i] > maks) {
            maks = nilai[i];
        }
        if (nilai[i] < min) {
            min = nilai[i];
        }
        total += nilai[i];
    }

    // Hitung rata-rata
    rata = (float) total / 5;

    // Tampilkan hasil
    printf("\nNilai tertinggi = %d\n", maks);
    printf("Nilai terendah  = %d\n", min);
    printf("Rata-rata nilai = %.2f\n", rata);

    return 0;
}

    • Ada 10 skor permainan yang disimpan dalam array: {85, 70, 90, 95, 60, 88, 76, 92, 100, 67}.

#include <stdio.h>

int main() {
    int skor[10] = {85, 70, 90, 95, 60, 88, 76, 92, 100, 67};

    // Tampilkan semua skor

    for (int i = 0; i < 10; i++) {
        printf("Skor ke-%d: %d\n", i + 1, skor[i]);
    }

    // Inisialisasi nilai terbesar dan terkecil
    terbesar = skor[0];
    terkecil = skor[0];

    // Proses mencari terbesar, terkecil, dan total

        if (skor[i] > terbesar) {
            terbesar = skor[i];
        }
        if (skor[i] < terkecil) {
            terkecil = skor[i];
        }
        total += skor[i];
    }

    // Hitung rata-rata

    // Tampilkan hasil
    printf("\n=== HASIL ANALISIS ===\n");
    printf("Skor terbesar : %d\n", terbesar);
    printf("Skor terkecil : %d\n", terkecil);
    printf("Rata-rata skor: %.2f\n", rata);

    return 0;

Latihan Praktek

  • SOAL 4: Bilangan Genap dan Ganjil
    • Diberikan array berisi 10 bilangan.
    • Memisahkan bilangan genap dan ganjil.
    • Menampilkan jumlah bilangan genap dan jumlah bilangan ganjil.

Contoh Program Bilangan Genap dan Ganjil

#include <stdio.h>

int main() {
    int bilangan[10];
    int genap[10], ganjil[10];
    int jumlahGenap = 0, jumlahGanjil = 0;

    // Input bilangan
    printf("Masukkan 10 bilangan:\n");
    for (int i = 0; i < 10; i++) {
        printf("Bilangan ke-%d: ", i + 1);
        scanf("%d", &bilangan[i]);
    }

    // Pisahkan genap dan ganjil
    for (int i = 0; i < 10; i++) {
        if (bilangan[i] % 2 == 0) {
            genap[jumlahGenap] = bilangan[i];
            jumlahGenap++;
        } else {
            ganjil[jumlahGanjil] = bilangan[i];
            jumlahGanjil++;
        }
    }

    // Tampilkan hasil
    printf("\nBilangan Genap: ");
    for (int i = 0; i < jumlahGenap; i++) {
        printf("%d ", genap[i]);
    }
    printf("\nJumlah bilangan genap: %d\n", jumlahGenap);

    printf("\nBilangan Ganjil: ");
    for (int i = 0; i < jumlahGanjil; i++) {
        printf("%d ", ganjil[i]);
    }
    printf("\nJumlah bilangan ganjil: %d\n", jumlahGanjil);

    return 0;
}

6.2 Array Dua Dimensi

Array dua dimensi adalah array yang memiliki baris dan kolom, seperti tabel atau matriks.

Deklarasi Array 2D

// Deklarasi array 2D
#include 

int main() {
    int matrix[2][3] = {
        {1, 2, 3},
        {4, 5, 6}
    };

    printf("Isi matrix:\n");

    for (int i = 0; i < 2; i++) {        // perulangan baris
        for (int j = 0; j < 3; j++) {    // perulangan kolom
            printf("%d ", matrix[i][j]);
        }
        printf("\n");  // pindah baris
    }

    return 0;
}

Input dan Output Array 2D

#include <stdio.h>

int main() {
    int matrix[3][3];
    int i, j;
    
    // Input matrix
    printf("Masukkan elemen matrix 3x3:\n");
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            printf("Elemen [%d][%d]: ", i, j);
            scanf("%d", &matrix[i][j]);
        }
    }
    
    // Tampilkan matrix
    printf("\nMatrix yang dimasukkan:\n");
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            printf("%d\t", matrix[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}

Penjumlahan Dua Matrix

#include <stdio.h>

int main() {
    int A[2][2] = {{1, 2}, {3, 4}};
    int B[2][2] = {{5, 6}, {7, 8}};
    int C[2][2];
    int i, j;
    
    // Penjumlahan matrix
    for (i = 0; i < 2; i++) {
        for (j = 0; j < 2; j++) {
            C[i][j] = A[i][j] + B[i][j];
        }
    }
    
    // Tampilkan hasil
    printf("Matrix A:\n");
    for (i = 0; i < 2; i++) {
        for (j = 0; j < 2; j++) {
            printf("%d\t", A[i][j]);
        }
        printf("\n");
    }
    
    printf("\nMatrix B:\n");
    for (i = 0; i < 2; i++) {
        for (j = 0; j < 2; j++) {
            printf("%d\t", B[i][j]);
        }
        printf("\n");
    }
    
    printf("\nHasil A + B:\n");
    for (i = 0; i < 2; i++) {
        for (j = 0; j < 2; j++) {
            printf("%d\t", C[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}

6.3 Pointer

Pointer adalah variabel yang menyimpan alamat memori dari variabel lain.

Deklarasi dan Penggunaan Pointer

#include <stdio.h>

int main() {
    int angka = 42;
    int *ptr;  // Deklarasi pointer
    
    ptr = &angka;  // ptr menunjuk ke alamat angka
    
    printf("Nilai angka: %d\n", angka);
    printf("Alamat angka: %p\n", &angka);
    printf("Nilai ptr: %p\n", ptr);
    printf("Nilai yang ditunjuk ptr: %d\n", *ptr);
    
    // Mengubah nilai melalui pointer
    *ptr = 100;
    printf("\nSetelah *ptr = 100:\n");
    printf("Nilai angka: %d\n", angka);
    printf("Nilai yang ditunjuk ptr: %d\n", *ptr);
    
    return 0;
}

Pointer dan Array

#include <stdio.h>

int main() {
    int arr[5] = {10, 20, 30, 40, 50};
    int *arr_ptr = arr;  // Nama array adalah pointer ke elemen pertama
    
    printf("Array menggunakan pointer:\n");
    for (int i = 0; i < 5; i++) {
        printf("arr[%d] = %d, *(arr_ptr + %d) = %d\n", 
               i, arr[i], i, *(arr_ptr + i));
    }
    
    return 0;
}

6.4 String

String adalah kumpulan karakter yang digunakan untuk menyimpan teks.

Deklarasi String

// String adalah array of characters
char nama[20] = "Budi";
char alamat[] = "Jakarta";

// Deklarasi tanpa inisialisasi
char kota[50];

Input dan Output String

#include <stdio.h>

int main() {
    char nama[50];
    
    // Input string tanpa spasi
    printf("Masukkan nama: ");
    scanf("%s", nama);
    printf("Nama: %s\n", nama);
    
    // Input string dengan spasi menggunakan fgets
    char alamat[100];
    printf("Masukkan alamat: ");
    getchar();  // Membersihkan buffer
    fgets(alamat, sizeof(alamat), stdin);
    printf("Alamat: %s", alamat);
    
    return 0;
}

Program C Menghitung Panjang String

#include <stdio.h>
#include <string.h>

int main() {
    char kalimat[100];

    printf("Program C Menghitung Panjang String\n");
    printf("-------------------------------------\n\n");

    // Input string
    printf("Masukkan String: ");
    fgets(kalimat, sizeof(kalimat), stdin);
    
    // Hapus newline dari fgets
    kalimat[strcspn(kalimat, "\n")] = 0;

    // Output panjang string
    printf("Panjang string adalah: %lu\n", strlen(kalimat));

    return 0;
}

Latihan Praktek

Topik: String

Tujuan: Menghitung panjang string

Soal: Budi ingin mengetahui berapa banyak karakter dalam pesan yang ia tulis untuk sahabatnya. Buatlah program C yang meminta Budi untuk memasukkan pesannya, kemudian menampilkan panjang pesan tersebut (jumlah karakter termasuk spasi).

Program Manipulasi String

#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
    char text[100];
    int hurufBesar = 0, hurufKecil = 0, angka = 0, spasi = 0;

    printf("Masukkan kalimat: ");
    fgets(text, sizeof(text), stdin);

    // Analisis karakter
    for (int i = 0; text[i] != '\0'; i++) {
        if (isupper(text[i])) {
            hurufBesar++;
        } else if (islower(text[i])) {
            hurufKecil++;
        } else if (isdigit(text[i])) {
            angka++;
        } else if (isspace(text[i])) {
            spasi++;
        }
    }

    printf("\n=== HASIL ANALISIS ===\n");
    printf("Huruf besar: %d\n", hurufBesar);
    printf("Huruf kecil: %d\n", hurufKecil);
    printf("Angka: %d\n", angka);
    printf("Spasi: %d\n", spasi);

    return 0;
}

Program Cek Palindrome

#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
    char kata[100];
    int panjang, isPalindrome = 1;

    printf("Masukkan kata: ");
    scanf("%s", kata);

    panjang = strlen(kata);

    // Ubah ke lowercase untuk perbandingan
    for (int i = 0; i < panjang; i++) {
        kata[i] = tolower(kata[i]);
    }

    // Cek palindrome
    for (int i = 0; i < panjang / 2; i++) {
        if (kata[i] != kata[panjang - 1 - i]) {
            isPalindrome = 0;
            break;
        }
    }

    if (isPalindrome) {
        printf("'%s' adalah palindrome\n", kata);
    } else {
        printf("'%s' bukan palindrome\n", kata);
    }

    return 0;
}
