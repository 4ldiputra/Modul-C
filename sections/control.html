Pernyataan if-else

Percabangan digunakan untuk membuat keputusan dalam program berdasarkan kondisi tertentu.

Struktur dasar if-else:
if (kondisi) {
    // Kode jika kondisi benar
} else {
    // Kode jika kondisi salah
}
Contoh if-else sederhana:
#include <stdio.h>

int main() {
    int angka;
    printf("Masukkan sebuah angka: ");
    scanf("%d", &angka);

    if (angka % 2 == 0) {
        printf("Angka genap\n");
    } else {
        printf("Angka ganjil\n");
    }

    return 0;
}

  • Buat program dalam bahasa C yang meminta pengguna memasukkan bilangan bulat.
  • Tentukan apakah bilangan tersebut positif atau negatif, lalu tampilkan hasilnya.

Pernyataan if-else if-else

Struktur dasar if-else if-else:
if (kondisi1) {
    // Kode jika kondisi1 benar
} else if (kondisi2) {
    // Kode jika kondisi2 benar
} else {
    // Kode jika semua kondisi salah
}
Contoh if-else if-else:
#include <stdio.h>

int main() {
    int num;
    printf("Masukkan sebuah angka: ");
    scanf("%d", &num);

    if (num > 0) {
        printf("Angka positif\n");
    } else if (num < 0) {
        printf("Angka negatif\n");
    } else {
        printf("Angka nol\n");
    }

    return 0;
}

Pernyataan switch-case

Switch case digunakan untuk percabangan multi-kondisi yang lebih efisien daripada if-else bertingkat.

Struktur dasar switch-case:
switch (ekspresi) {
    case nilai1:
        // Kode jika ekspresi == nilai1
        break;
    case nilai2:
        // Kode jika ekspresi == nilai2
        break;
    default:
        // Kode jika tidak ada case yang cocok
}
Contoh switch-case sederhana:
#include <stdio.h>

int main() {
    int pilihan;
    printf("Pilih menu (1-3): ");
    scanf("%d", &pilihan);

    switch (pilihan) {
        case 1:
            printf("Anda memilih menu 1\n");
            break;
        case 2:
            printf("Anda memilih menu 2\n");
            break;
        case 3:
            printf("Anda memilih menu 3\n");
            break;
        default:
            printf("Pilihan tidak valid\n");
    }

    return 0;
}
Contoh switch-case dengan if-else:
#include <stdio.h>

int main() {
    int menu, nilai;
    printf("Pilih mata pelajaran (1=Matematika, 2=Bahasa): ");
    scanf("%d", &menu);

    switch (menu) {
        case 1:
            printf("Masukkan nilai Matematika: ");
            scanf("%d", &nilai);
            if (nilai >= 75) {
                printf("Lulus Matematika\n");
            } else {
                printf("Tidak lulus Matematika\n");
            }
            break;
        case 2:
            printf("Masukkan nilai Bahasa: ");
            scanf("%d", &nilai);
            if (nilai >= 70) {
                printf("Lulus Bahasa\n");
            } else {
                printf("Tidak lulus Bahasa\n");
            }
            break;
        default:
            printf("Pilihan tidak valid\n");
    }

    return 0;
}

  • Buat program simulasi ATM dengan menu:
    1. Cek Saldo 2. Tarik Tunai 3. Setor Tunai 4. Keluar
    Gunakan percabangan untuk menampilkan hasil sesuai menu yang dipilih.
    Saldo awal misalnya Rp1.000.000.
Contoh switch-case dengan if-else:
#include 

int main() {
    int saldo = 1000000; // saldo awal Rp1.000.000
    int pilihan, jumlah;

    printf("\n=== MENU ATM ===\n");
    printf("1. Cek Saldo\n");
    printf("2. Tarik Tunai\n");
    printf("3. Setor Tunai\n");
    printf("4. Keluar\n");
    printf("Pilih menu: ");
    scanf("%d", &pilihan);

    switch (pilihan) {
        case 1:
            printf("Saldo Anda: Rp%d\n", saldo);
            break;
        case 2:
            printf("Masukkan jumlah tarik: Rp");
            scanf("%d", &jumlah);
            if (jumlah <= saldo) {
                saldo -= jumlah;
                printf("Tarik tunai berhasil. Saldo sekarang: Rp%d\n", saldo);
            } else {
                printf("Saldo tidak cukup!\n");
            }
            break;
        case 3:
            printf("Masukkan jumlah setor: Rp");
            scanf("%d", &jumlah);
            if (jumlah > 0) {
                saldo += jumlah;
                printf("Setor tunai berhasil. Saldo sekarang: Rp%d\n", saldo);
            } else {
                printf("Jumlah setor harus lebih dari 0!\n");
            }
            break;
        case 4:
            printf("Terima kasih telah menggunakan ATM.\n");
            break;
        default:
            printf("Pilihan tidak valid!\n");
    }

    return 0;
}
