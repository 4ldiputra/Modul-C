7.1 Definisi dan Penggunaan Struct

Struct adalah tipe data buatan pengguna yang memungkinkan pengelompokan beberapa variabel dengan tipe data berbeda menjadi satu kesatuan logis.

Membuat Struktur Data Sederhana

Struct cocok untuk merepresentasikan entitas dunia nyata, seperti:

  1. Data buku: judul, pengarang, penerbit, tahun terbit
  2. Data mahasiswa: nama, NIM, jurusan
  3. Data karyawan: nama, ID, gaji

Deklarasi Struct

#include 
#include 

struct Buku {
    char judul[100];
    char pengarang[100];
    char penerbit[100];
    int tahun_terbit;
};

Mengisi dan Mengakses Anggota Struct

struct Buku buku1;

// Mengisi data
strcpy(buku1.judul, "Pemrograman C untuk Pemula");
strcpy(buku1.pengarang, "Ahmad Fauzi");
strcpy(buku1.penerbit, "Penerbit Informatika");
buku1.tahun_terbit = 2022;

// Menampilkan data
printf("Judul: %s\n", buku1.judul);
printf("Tahun: %d\n", buku1.tahun_terbit);

Inisialisasi Struct

// Saat deklarasi
struct Buku buku2 = {"Dasar C", "Budi Santoso", "Erlangga", 2020};

// Atau dengan penugasan field (C99+)
struct Buku buku3 = {
    .judul = "Struktur Data",
    .pengarang = "Dian Pratiwi",
    .penerbit = "Andi Publisher",
    .tahun_terbit = 2021
};

Contoh Program Dasar Struct

#include 
#include 

struct Mahasiswa {
    char nama[50];
    char nim[15];
    float ipk;
};

int main() {
    struct Mahasiswa mhs1;
    
    strcpy(mhs1.nama, "Rina Wijaya");
    strcpy(mhs1.nim, "23060123");
    mhs1.ipk = 3.75;
    
    printf("Nama: %s\n", mhs1.nama);
    printf("NIM : %s\n", mhs1.nim);
    printf("IPK : %.2f\n", mhs1.ipk);
    
    return 0;
}

7.2 Array dari Struct

Kita bisa membuat array berisi banyak struct untuk menyimpan data kolektif.

Contoh: Daftar Mahasiswa

#include 
#include 

struct Mahasiswa {
    char nama[50];
    char nim[15];
    float ipk;
};

int main() {
    struct Mahasiswa kelas[3] = {
        {"Ali", "23060101", 3.5},
        {"Budi", "23060102", 3.8},
        {"Citra", "23060103", 3.9}
    };
    
    printf("Daftar Mahasiswa:\n");
    for (int i = 0; i < 3; i++) {
        printf("%d. %s (%s) - IPK: %.2f\n",
               i+1, kelas[i].nama, kelas[i].nim, kelas[i].ipk);
    }
    
    return 0;
}

Input Data ke Array Struct dari User

#include 
#include 

struct Mahasiswa {
    char nama[50];
    char nim[15];
    float ipk;
};

int main() {
    struct Mahasiswa mhs[3];
    
    for (int i = 0; i < 3; i++) {
        printf("Data mahasiswa ke-%d:\n", i+1);
        printf("Nama: ");
        fgets(mhs[i].nama, sizeof(mhs[i].nama), stdin);
        mhs[i].nama[strcspn(mhs[i].nama, "\n")] = 0; // hapus newline
        
        printf("NIM: ");
        fgets(mhs[i].nim, sizeof(mhs[i].nim), stdin);
        mhs[i].nim[strcspn(mhs[i].nim, "\n")] = 0;
        
        printf("IPK: ");
        scanf("%f", &mhs[i].ipk);
        getchar(); // bersihkan buffer
        printf("\n");
    }
    
    printf("=== DAFTAR MAHASISWA ===\n");
    for (int i = 0; i < 3; i++) {
        printf("Nama: %s\n", mhs[i].nama);
        printf("NIM : %s\n", mhs[i].nim);
        printf("IPK : %.2f\n\n", mhs[i].ipk);
    }
    
    return 0;
}

7.3 Struct Bersarang (Nested Struct)

Struct bisa berisi struct lain untuk merepresentasikan data bertingkat.

Contoh: Data Karyawan dengan Alamat

struct Alamat {
    char jalan[100];
    char kota[50];
    char kode_pos[10];
};

struct Karyawan {
    char nama[50];
    int id;
    float gaji;
    struct Alamat alamat; // nested struct
};

Penggunaan Nested Struct

#include 
#include 

struct Alamat {
    char jalan[100];
    char kota[50];
    char kode_pos[10];
};

struct Karyawan {
    char nama[50];
    int id;
    float gaji;
    struct Alamat alamat;
};

int main() {
    struct Karyawan kary;
    
    printf("Nama: ");
    fgets(kary.nama, sizeof(kary.nama), stdin);
    kary.nama[strcspn(kary.nama, "\n")] = 0;
    
    printf("ID: ");
    scanf("%d", &kary.id);
    printf("Gaji: ");
    scanf("%f", &kary.gaji);
    getchar(); // bersihkan buffer
    
    printf("Alamat:\n");
    printf("Jalan: ");
    fgets(kary.alamat.jalan, sizeof(kary.alamat.jalan), stdin);
    kary.alamat.jalan[strcspn(kary.alamat.jalan, "\n")] = 0;
    
    printf("Kota: ");
    fgets(kary.alamat.kota, sizeof(kary.alamat.kota), stdin);
    kary.alamat.kota[strcspn(kary.alamat.kota, "\n")] = 0;
    
    printf("Kode Pos: ");
    fgets(kary.alamat.kode_pos, sizeof(kary.alamat.kode_pos), stdin);
    kary.alamat.kode_pos[strcspn(kary.alamat.kode_pos, "\n")] = 0;
    
    printf("\n=== DATA KARYAWAN ===\n");
    printf("Nama: %s\n", kary.nama);
    printf("ID: %d\n", kary.id);
    printf("Gaji: Rp %.2f\n", kary.gaji);
    printf("Alamat: %s, %s %s\n",
           kary.alamat.jalan, kary.alamat.kota, kary.alamat.kode_pos);
    
    return 0;
}

Latihan Praktek

  • SOAL 1: Data Mahasiswa & Kontak
    • Buat struct Kontak dengan field: email (char[50]), noHP (char[15]).
    • Buat struct Mahasiswa dengan: nama, nim, jurusan, dan kontak (bertipe struct Kontak).
    • Input data 1 mahasiswa dari user.
    • Tampilkan kembali datanya.

7.4 Pencarian dalam Array Struct

Contoh: Cari Produk Berdasarkan Kode

#include 
#include 

struct Produk {
    char nama[50];
    char kode[10];
    float harga;
    int stok;
};

int main() {
    struct Produk daftar[5] = {
        {"Laptop", "LP001", 15000000, 10},
        {"Mouse", "MS001", 150000, 25},
        {"Keyboard", "KB001", 500000, 15},
        {"Monitor", "MN001", 2500000, 8},
        {"Speaker", "SP001", 750000, 12}
    };
    
    char cari[10];
    printf("Cari kode produk: ");
    scanf("%s", cari);
    
    int ditemukan = 0;
    for (int i = 0; i < 5; i++) {
        if (strcmp(daftar[i].kode, cari) == 0) {
            printf("\nProduk ditemukan!\n");
            printf("Nama  : %s\n", daftar[i].nama);
            printf("Harga : Rp %.0f\n", daftar[i].harga);
            printf("Stok  : %d\n", daftar[i].stok);
            ditemukan = 1;
            break;
        }
    }
    
    if (!ditemukan) {
        printf("Produk tidak ditemukan.\n");
    }
    
    return 0;
}

Latihan Praktek

  • SOAL 2: Data Buku Perpustakaan
    • Buat struct Buku dengan: judul, pengarang, isbn, tahun.
    • Buat array untuk 10 buku.
    • Input dari user.
    • Tampilkan hanya buku dengan tahun > 2010.

7.5 Struct dengan Array di Dalamnya

Contoh: Nilai Ujian Siswa

#include 
#include 

struct Siswa {
    char nama[50];
    float nilai[5]; // array dalam struct
};

int main() {
    struct Siswa siswa;
    float total = 0;
    
    printf("Nama siswa: ");
    fgets(siswa.nama, sizeof(siswa.nama), stdin);
    siswa.nama[strcspn(siswa.nama, "\n")] = 0;
    
    printf("Masukkan 5 nilai:\n");
    for (int i = 0; i < 5; i++) {
        printf("Nilai %d: ", i+1);
        scanf("%f", &siswa.nilai[i]);
        total += siswa.nilai[i];
    }
    
    float rata = total / 5;
    
    printf("\n=== RATA-RATA NILAI ===\n");
    printf("Nama: %s\n", siswa.nama);
    printf("Rata-rata: %.2f\n", rata);
    
    return 0;
}

Tips Penggunaan Struct di C:

  • Gunakan strcpy() untuk menyalin string ke field char[].
  • Gunakan strcmp() untuk membandingkan string.
  • Hindari scanf("%s") untuk input string ber-spasi ‚Äî gunakan fgets().
  • Bersihkan buffer newline dengan getchar() atau strcspn().
  • Struct tidak bisa langsung dicetak/dibandingkan ‚Äî akses field-nya satu per satu.

Kapan Menggunakan Struct? Gunakan struct saat kamu perlu mengelompokkan data heterogen (berbeda tipe) yang merepresentasikan satu entitas, seperti orang, produk, transaksi, atau dokumen.

7.6 Union

Union adalah tipe data buatan pengguna yang mirip dengan struct, tetapi semua anggotanya berbagi lokasi memori yang sama. Hanya satu anggota union yang dapat menyimpan nilai pada satu waktu.

Deklarasi Union

#include 

union Data {
    int angka;
    float pecahan;
    char teks[20];
};

Perbedaan Struct vs Union

  • Struct: Setiap anggota punya memori sendiri ‚Üí total ukuran = jumlah semua anggota.
  • Union: Semua anggota berbagi memori yang sama ‚Üí total ukuran = ukuran anggota terbesar.
  • Di union, hanya satu nilai valid pada satu waktu. Menulis ke anggota lain akan menimpa nilai sebelumnya.

Contoh Penggunaan Union

#include 
#include 

union Data {
    int angka;
    float pecahan;
    char teks[20];
};

int main() {
    union Data data;

    // Simpan nilai integer
    data.angka = 100;
    printf("Nilai angka: %d\n", data.angka);

    // Simpan nilai float ‚Üí nilai integer HILANG!
    data.pecahan = 3.14f;
    printf("Nilai pecahan: %.2f\n", data.pecahan);
    printf("Nilai angka sekarang: %d (tidak valid!)\n", data.angka);

    // Simpan string
    strcpy(data.teks, "C Union");
    printf("Teks: %s\n", data.teks);

    // Cetak ukuran union
    printf("\nUkuran union: %zu byte\n", sizeof(data)); // = 20 (karena char[20])

    return 0;
}

Kapan Menggunakan Union?

Union berguna saat kamu ingin menyimpan salah satu dari beberapa tipe data dalam satu variabel, misalnya:

  • Membaca data biner yang bisa berupa int, float, atau char.
  • Menghemat memori dalam sistem embedded.
  • Membuat parser atau interpreter sederhana.

Contoh Praktis: Simpan Tipe Data Dinamis

#include 

// Enum untuk melacak tipe data yang aktif
enum Tipe { INTEGER, FLOAT };

// Union + struct untuk menyimpan nilai + tipenya
struct NilaiDinamis {
    enum Tipe tipe;
    union {
        int i;
        float f;
    } data;
};

int main() {
    struct NilaiDinamis v;

    // Simpan integer
    v.tipe = INTEGER;
    v.data.i = 42;
    if (v.tipe == INTEGER) {
        printf("Nilai: %d (integer)\n", v.data.i);
    }

    // Ganti ke float
    v.tipe = FLOAT;
    v.data.f = 2.71f;
    if (v.tipe == FLOAT) {
        printf("Nilai: %.2f (float)\n", v.data.f);
    }

    return 0;
}

Latihan Praktek

  • SOAL: Buat program yang menggunakan union untuk menyimpan:
    • Sebuah bilangan bulat (int)
    • Sebuah karakter (char)
    • Sebuah bilangan pecahan (float)
    Program harus:
    1. Meminta user memilih tipe data yang ingin dimasukkan.
    2. Input nilai sesuai pilihan.
    3. Tampilkan nilai tersebut.
    4. Cetak juga ukuran union tersebut.
